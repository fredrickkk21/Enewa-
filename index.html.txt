/* PARALLAX STARS */
function createStars(layer,count){
  for(let i=0;i<count;i++){
    let star=document.createElement("div");
    star.className="star";
    star.style.width=Math.random()*3+"px";
    star.style.height=star.style.width;
    star.style.top=Math.random()*100+"%";
    star.style.left=Math.random()*100+"%";
    layer.appendChild(star);
  }
}
createStars(document.getElementById("starsBack"),80);
createStars(document.getElementById("starsFront"),40);

/* PROPOSAL */
const music=document.getElementById("music");

function propose(){
  music.play().catch(()=>{});
  document.getElementById("proposal").style.display="block";
  fireworks();
}

/* FIREWORKS */
function fireworks(){
  for(let i=0;i<60;i++){
    let f=document.createElement("div");
    f.className="firework";
    f.style.left="50%";
    f.style.top="50%";
    f.style.setProperty("--x",(Math.random()*400-200)+"px");
    f.style.setProperty("--y",(Math.random()*400-200)+"px");
    document.body.appendChild(f);
    setTimeout(()=>f.remove(),1200);
  }
}
